输入描述:

输入一行字符串，可以有空格

统计其中英文字符，空格字符，数字字符，其他字符的个数

#include <iostream>
using namespace std;
  
int main()
{
    string s;
    
    while(getline(cin, s))
    {
        int cnt1 = 0, cnt2 = 0, cnt3 = 0, cnt4 = 0;
        for(int i = 0;i < s.size(); i ++)
        {
            if((s[i] >= 'a' && s[i] <= 'z') || (s[i] >= 'A' && s[i] <= 'Z'))
                cnt1 ++;
            else if(s[i] == ' ')
                cnt2 ++;
            else if(s[i] >= '0' && s[i] <= '9')
                cnt3 ++;
            else
                cnt4 ++;
        }
        cout << cnt1 << endl << cnt2 << endl << cnt3 << endl << cnt4 << endl;
    }
    return 0;
}
