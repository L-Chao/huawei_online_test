一个长整数。

#include<iostream>
#include<string>
using namespace std;

int main()
{
    unsigned int a, b, c, d, shi;//10.0.3.193
    //int 32 1正负数
    //long a, b, c, d, shi;
    char dot;
    while(cin >> a >> dot >> b >> dot >> c >> dot >> d)
    {
        cout << (a << 24) + (b << 16) + (c << 8) + d << endl;
        
        //输入十进制
        cin >> shi;
        cout << ((shi & 0xff000000) >> 24) << '.'<< ((shi & 0x00ff0000) >> 16) << '.' << ((shi & 0x0000ff00) >> 8) << '.' << (shi & 0x000000ff) << endl;  
    }

}
